


CREATE DATABASE 'note-a-list';
CREATE TABLE 'List' (
	id INT,
	name VARCHAR(100)
	key(id)
	);
CREATE TABLE 'List Relationship' (
	parent_id INT,
	child_id INT,
	foreign_key(parent_id),
	foreign_key(child_id)
	);
CREATE TABLE 'Note Relationship' (
	list_id INT,
	note_id INT,
	foreign_key(list_id),
	foreign_key(note_id)
	)
List (id, name) id -> name (key = id)

List Relationship (parent_id, child_id) foreign_key(parent_id) foreign_key(child_id) 
Note Relationship (list_id, note_id) foreign_key(list_id) foreign_key(note_id)
Note Details (note_id, title, date, description) note_id -> title, date, description (key = note_id)
